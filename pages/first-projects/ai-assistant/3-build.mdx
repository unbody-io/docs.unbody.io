# Build An AI Assistant

<Tabs size={"sm"} variant={"underlined"} className={"mt-2"}>
    <Tab title={"SDK (.ts)"}>
    ```bash
    npm i unbody
    ```
    ```typescript
    import { Unbody } from 'unbody';

    const unbody = new Unbody({
      projectId: 'your-project-id',
      apiKey: 'your-api-key'
    });
    
    // Create assistant
    const assistant = await unbody.assistant.create({
      name: 'Community Helper',
      description: 'Helps answer community questions',
      options: {
        personality: 'helpful and friendly',
        contextStrategy: 'semantic'
      }
    });

    // Chat with context
    const response = await unbody.assistant
      .chat(assistant.id)
      .withMessage('What were the main topics discussed last week?')
      .withContext({
        timeframe: '7d',
        relevanceThreshold: 0.8
      })
      .exec();

    // Stream responses
    const stream = await unbody.assistant
      .chat(assistant.id)
      .withMessage('Summarize recent discussions')
      .stream()
      .exec();

    for await (const chunk of stream) {
      process.stdout.write(chunk.content);
    }
    ```
